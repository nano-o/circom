Theory IsZero

theory IsZero
  imports Main
begin

text ‹Proof that the IsZero template is correct, assuming the signals are values of some arbitrary field›

definition is_zero :: "'a::field ⇒ 'a::field" where
  "is_zero in_sig ≡ if in_sig = 0 then 1 else 0"

text ‹First we show that, if the constraints are satisfied, then the output signal is correct.›

lemma l1:
  fixes in_sig inv_sig out_sig :: "'a::field"
  defines "out_sig ≡ (-in_sig)*inv_sig + 1"
  assumes "in_sig*out_sig = 0"
  shows "out_sig = is_zero in_sig"
    ― ‹note that @{term inv_sig} is left unconstrained›
  by (metis add_0 assms(1,2) is_zero_def mult_eq_0_iff mult_minus_left)

text ‹Next we show that the expression assigned to the inv signal satisfies the constraints›

lemma l2:
  fixes in_sig inv_sig out_sig :: "'a::field"
  defines "inv_sig ≡ (if in_sig ≠ 0 then (1/in_sig) else 0)"
  and "out_sig ≡ (-in_sig)*inv_sig + 1"
shows "in_sig*out_sig = 0"
  by (simp add: inv_sig_def out_sig_def)

end
